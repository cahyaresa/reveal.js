The Difference
Define Variable
										
--> SASS
$font-size: 16px;
	div {
		font-size: $font-size;
	}
--> LESS
@font-size: 16px;
	div {
		font-size: @font-size;
	}
--> STYLUS
font-size = 16px;
	div {
		font-size font-size;
	}
-->CSS3
	div {
		font-size : 16px;
	}

Nesting/ Hierarchy
											
	--> SASS

	$link-hover: #229ed3;

		margin: 0;

			float: left;

			color: $link-color;

	@link-hover: #229ed3;

		margin: 0;
	
		li {
			float: left;
		}
	
		a {
			color: @link-color;
	
			&:hover {
				color: @link-hover;
			}
		}
	}
	--> STYLUS
	link-color = #999
	link-hover = #229ed3
